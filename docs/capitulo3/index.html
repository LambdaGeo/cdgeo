def load_dataframe_by_urls(urls, cols):
  df_total = pd.DataFrame() # um dataframe vazio
  for url in urls:
    print ("URL:", url)
    df = pd.read_csv(url, sep=";", usecols=cols)
    df_total = pd.concat([df_total, df])
  return df_total
